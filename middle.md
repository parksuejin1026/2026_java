import java.io.*; // 파일 장치 및 문자 입출력 스트림(FileWriter, BufferedReader 등) 사용을 위해 패키지를 가져옵니다.
import java.util.Iterator; // 컬렉션 내부 요소를 인덱스 없이 표준화된 방식으로 순회하기 위한 인터페이스를 가져옵니다.
import java.util.Random; // 좀비의 무작위 이동 및 보물의 랜덤 위치 선정을 위해 난수 생성 클래스를 가져옵니다.
import java.util.Scanner; // 콘솔창으로부터 유저의 메뉴 입력(정수)을 받기 위해 스캐너 클래스를 가져옵니다.
import java.util.Vector; // 획득한 보물들을 순서대로 저장하고 동기화를 보장하는 가변 크기 배열 컬렉션을 가져옵니다.

// ==========================================
// [문제 2 & 3] 좀비 게임 메인 컨트롤러 및 PPT 교안 준수 문자 스트림부
// ==========================================
public class ZombieGame {
    // 세이브 데이터가 물리적으로 저장될 경로를 상수(final) 문자열로 지정합니다.
    private static final String FILE_PATH = "C:\\temp\\zombie.sav";
    
    // Zombie 클래스(다른 클래스)에서 가방에 실시간 접근 및 수집 판정을 내릴 수 있도록 패키지(default) 제한 범위로 선언합니다.
    // Gem 클래스를 요소(Element) 타입으로 구체화한 가변 크기 Vector 컬렉션을 생성합니다.
    static Vector<Gem> inventory = new Vector<>(); 
    
    private static Human player; // 게임 내 주인공 플레이어를 제어할 Human 참조 변수입니다.
    private static Zombie zombie1; // 독립적으로 스레드 구동되어 살아 움직일 첫 번째 좀비 객체 변수입니다.
    private static Zombie zombie2; // 동시에 백그라운드에서 플레이어를 추격할 두 번째 좀비 객체 변수입니다.
    private static Gem currentGem; // 현재 맵상(1~20칸 범위)에 놓여있는 단 하나의 보물 타깃 객체입니다.
    private static Random r = new Random(); // 다양한 난수 공식(보물 종류, 보물 좌표)을 뽑아낼 도구입니다.
    
    // 자바 자이로스코프이자 프로그램의 핵심 시작점인 메인 스레드 메소드입니다.
    public static void main(String[] args) {
        // 키보드 콘솔 입력을 안전하게 읽어 들이기 위한 Scanner 인스턴스를 생성합니다.
        Scanner sc = new Scanner(System.in);
        
        // C:\temp 폴더 경로가 하드디스크에 존재하는지 검증하기 위해 물리 File 객체를 껍데기로 생성합니다.
        File tempDir = new File("C:\\temp"); 
        
        // 조건문: 만약 지정된 경로의 물리 폴더(temp)가 컴퓨터에 실재하지 않는 청정 상태라면
        if (!tempDir.exists()) { 
            tempDir.mkdir();  // 파일 시스템 명령을 통해 C 드라이브 직하에 temp 이름의 폴더를 물리 생성합니다.
        }
        
        // 게임의 화려한 개막을 알리는 오프닝 웰컴 타이틀을 콘솔에 찍어줍니다.
        System.out.println("========= 서바이벌 좀비 보물 게임을 시작합니다 =========");
        
        // "플레이어 1"이라는 고유 명칭을 부여하여 주인공 Human 인스턴스를 생성합니다.
        player = new Human("플레이어 1");
        
        // 첫 번째 좀비를 20번 칸에 배치하고 플레이어의 심장 위치를 추적할 수 있도록 객체를 엮어 생성합니다.
        zombie1 = new Zombie(20, "좀비 1", player); 
        // 두 번째 좀비는 10번 칸에 수동 배치하여 양방향 압박 수비를 형성하도록 설정합니다.
        zombie2 = new Zombie(10, "좀비 2", player);
        
        // 상속받은 Thread 클래스의 start()를 호출하여 좀비 1호의 독립 제어 흐름을 백그라운드에 등록 및 구동합니다.
        zombie1.start(); 
        // 동시에 좀비 2호의 독립 run() 루프 스케줄러를 개통하여 멀티스레드 환경을 완전 구축합니다.
        zombie2.start(); 
        
        // 게임 세상이 열렸으므로 필드 상에 무작위로 첫 번째 보물을 드롭(생성)합니다.
        generateNewGem(); 
        
        // 유저가 강제로 종료(7)하거나 좀비에게 물리 전사하기 전까지 메인 제어 루프를 무한 가동합니다.
        while (true) {
            // 메인 액션 턴 직전에 전술적 판단을 내릴 수 있도록 실시간 위치 판독 현황판을 인쇄합니다.
            System.out.println("\n==================================================");
            // 현재 플레이어의 좌표와 독립 움직이던 좀비들의 실시간 좌표 상태를 보기 좋게 포맷팅 출력합니다.
            System.out.printf("[현재 위치] 플레이어: %d | 좀비1: %d | 좀비2: %d\n", 
                              player.getPos(), zombie1.getPos(), zombie2.getPos());
            System.out.println("==================================================");
            
            // 사용자에게 행동 지시 선택지 가이드를 화면에 텍스트로 제시합니다.
            System.out.print("[메뉴] 왼쪽(1), 오른쪽(2), 점프(3), 획득한 보물(4), 저장하기(5), 이어하기(6), 종료(7) >> ");
            
            int menu = 0; // 사용자가 선택한 메뉴 번호를 담을 임시 저장고 변수입니다.
            
            // 조건문: 스캐너 버퍼에 들어온 유저의 키보드 입력값이 오염되지 않은 온전한 '정수' 형태일 때만 발동합니다.
            if (sc.hasNextInt()) { 
                menu = sc.nextInt(); // 정수 값을 명확하게 읽어와 menu 변수에 적재합니다.
            } else { // 만약 유저가 문자나 이상한 기호를 잘못 타이핑했다면
                sc.next(); // 스캐너 입력 버퍼에 쌓인 잘못된 쓰레기 값을 청소하여 비워냅니다.
                continue; // 아래의 위험한 스위치 문을 건너뛰고 바로 while문의 맨 처음(행동판 출력)으로 제어를 되돌립니다.
            }
            
            // 검증이 완료된 올바른 메뉴 번호를 기반으로 실제 동작 제어 분기를 수행합니다.
            switch (menu) {
                case 1: player.moveLeft(); break;  // 1번: 주인공을 왼쪽 좌표계 방향(-1)으로 이동시킵니다.
                case 2: player.moveRight(); break; // 2번: 주인공을 오른쪽 좌표계 방향(+1)으로 이동시킵니다.
                case 3: player.jump(); break;      // 3번: 좀비를 뛰어넘을 수 있는 도약 점프 로직을 작동합니다.
                case 4: printInventory(); break;   // 4번: 현재까지 가방(Vector)에 누적 수집한 보물 목록을 보여줍니다.
                case 5: saveGame(); break;         // 5번: 현재의 보물 인벤토리 자산을 텍스트 세이브 파일로 기록 보관합니다.
                case 6: loadGame(); break;         // 6번: 과거에 백업해 둔 저장 파일 데이터를 읽어와 복원(이어하기)합니다.
                case 7: // 7번: 수동 안전 종료 분기입니다.
                    System.out.println("게임이 종료되었습니다.");
                    zombie1.stopZombie(); // 안전 종료를 위해 열려있던 좀비 1호 스레드 무한 루프 플래그를 정지시킵니다.
                    zombie2.stopZombie(); // 좀비 2호 스레드 또한 연쇄적으로 무한 제어를 중단시켜 시스템 자원을 반환합니다.
                    sc.close(); // 사용이 끝난 콘솔 스캐너 리소스 스트림을 닫아 메모리 누수를 완벽 차단합니다.
                    return; // 메인 메소드 자체를 즉시 완전 탈출(프로그램 완전 종료)시킵니다.
                default: // 1~7 범위를 벗어난 기이한 숫자가 입력되었을 때 제어하는 디폴트 안전장치입니다.
                    System.out.println("잘못된 메뉴 선택입니다.");
                    continue; // 알림을 준 후 즉시 while문 상단으로 루프 턴을 넘깁니다.
            }
            
            // 플레이어가 수동으로 움직인 직후, 재수 없게 좀비가 선점하고 있던 소굴 좌표로 기어 들어갔는지 충돌 체크를 진행합니다.
            // 조건문: 플레이어의 위치가 좀비1의 위치와 겹치거나, 혹은 좀비2의 위치와 단 1칸이라도 일치하게 되는 긴급 조우 상황입니다.
            if (player.getPos() == zombie1.getPos() || player.getPos() == zombie2.getPos()) { 
                System.out.println("\n☠ 이동 또는 점프한 위치에 좀비가 대기하고 있었습니다! GAMEOVER!"); 
                printFinalScore(); // 게임이 잔혹하게 끝났으므로 수집 리포트 요약 통계를 최종 화면에 명시합니다.
                zombie1.stopZombie(); // 혼자 돌며 리소스를 갉아먹을 좀비 1호의 백그라운드 스레드 엔진을 강제 다운시킵니다.
                zombie2.stopZombie(); // 좀비 2호의 백그라운드 스레드 무한 엔진 또한 안전하게 다운 세팅합니다.
                sc.close(); // 스캐너 장치 참조를 안전하게 해제하여 반환합니다.
                return; // 메인 실행 스레드 자체를 반환 종결하여 게임을 완전히 폭파(종료)합니다.
            }
            
            // 좀비와 충돌하지 않고 무사히 살아남았다면, 혹시 그 자리가 보물이 떨어진 명당자리인지 수확 검사를 수행합니다.
            checkGemCollision(); 
        }                
    }
    
    /**
     * 필드 위에 무작위 보석 종류와 무작위 드롭 좌표를 계산하여 필드 기물을 최신화하는 공장 메소드입니다.
     */
    private static void generateNewGem() { 
        int type = r.nextInt(3) + 1; // 1, 2, 3 중에서 하나의 주사위 숫자를 임의로 선택합니다.
        
        // 삼항연산자 이중 구조: 뽑힌 정수가 1이면 "다이아", 2이면 "금", 그것도 아니면 최종적으로 "은" 변형 이름을 부여합니다.
        String gemName = (type == 1) ? "다이아" : (type == 2) ? "금" : "은"; 
        
        // 보물이 숨겨질 물리 좌표 구역을 1번부터 20번 격자 칸 사이에서 무작위로 확정합니다.
        int gemPos = r.nextInt(20) + 1; 
        
        // 문제 1번 규격의 초기화 생성자(Gem("금", 2, 10) 형태) 설계 공식에 맞춰 필드 보물 인스턴스를 새롭게 빌드합니다.
        currentGem = new Gem(gemName, type, gemPos); 
        
        // 신규 보석이 어디 숨어있는지 필드 알림 텍스트 문장을 콘솔 화면에 상세 인쇄합니다.
        System.out.println("\n [신규 보물 출현] " + gemName + "이(가) " + gemPos + "번 위치에 생성되었습니다."); 
    }

    /**
     * 플레이어의 좌표와 필드 보석의 좌표가 정합을 이루어 수확 판정이 났는지 체크하는 메소드입니다.
     */
    private static void checkGemCollision() {
        // 조건문: 플레이어가 현재 밟고 서 있는 디딤돌 좌표와 필드에 생성해둔 보석 기물의 배치 좌표가 완벽하게 맞물린 상황입니다.
        if (player.getPos() == currentGem.getPos()) { 
            currentGem.getItem();  // Gem 객체 내부에 설계된 "다이아(금,은)을 획득했습니다." 공용 양식 메시지를 출력 유도합니다.
            inventory.add(currentGem);  // 검증 완료된 실물 보물 인스턴스 참조를 수집 가방 컬렉션(Vector) 맨 끝 슬롯에 밀어 넣어 저장합니다.
            generateNewGem(); // 수확이 끝나 빈자리가 되었으므로 즉시 다음 수집 경쟁을 유도할 새로운 보물을 맵에 드롭합니다.
        }
    }

    /**
     * [제네릭과 컬렉션 PPT 방식] Iterator 표준 인터페이스 규칙에 맞춰 수집 인벤토리를 투명하게 공개하는 출력부입니다.
     */
    private static void printInventory() {
        System.out.println("\n[현재 가방 속 보물 목록]"); 
        // 예외 필터: 인벤토리 가방(Vector 컬렉션) 내부 요소 개수가 완전히 제로(0)인 텅 빈 상태인지를 검사합니다.
        if (inventory.isEmpty()) { 
            System.out.println("가방이 텅 비어있습니다."); // 알림을 인쇄합니다.
            return; // 훑을 데이터 대상이 전무하므로 함수 연산을 즉시 중단하고 반환 제어 처리합니다.
        }
        
        // [PPT 표준 방식] Vector 컬렉션 인스턴스로부터 데이터를 처음부터 일렬로 훑어줄 전용 이터레이터 로봇 개체를 획득합니다.
        Iterator<Gem> it = inventory.iterator();
        
        // 중간고사 문제지 요구 문구 양식 서두인 총 누적 획득 개수를 먼저 화면에 밀어 출력합니다.
        System.out.print("현재 총 " + inventory.size() + "개 ");
        
        // PPT 예제 방식: hasNext()를 호출해 "로봇아, 다음 칸에 꺼내올 보물 객체가 실존하니?"라고 물어보며 무한 루프를 가동합니다.
        while(it.hasNext()) {
            Gem g = it.next(); // true 판정이 났으므로 다음 칸으로 전진하여 보물(Gem) 인스턴스의 실체 참조를 꺼내어 변수 g에 이관합니다.
            System.out.print(g.getName()); // 꺼내온 보물의 텍스트 순수 명칭("다이아" 등)을 줄 바꿈 없이 화면에 출력합니다.
            
            // 문장 부호 제어 디테일: 만약 내 다음 칸에 대기 중인 출력 대상 보물이 아직 더 누적되어 남아있다면
            if(it.hasNext()) {
                System.out.print(", "); // 이름 간 경계를 명확히 구분 지어줄 이쁜 콤마 기호를 중간에 삽입 인쇄합니다.
            }
        }
        // 이터레이터가 마지막 데이터까지 완벽 스캔을 마쳐 while 루프가 깨지면, 꼬리말 마무리 서식을 찍고 줄바꿈 마감합니다.
        System.out.println("을 가지고 있습니다.");
    }
    
    /**
     * 좀비에게 처참히 물려 죽었을 때 게임 오버 시점의 수집 이력을 명징하게 정리 리포트해주는 스태틱 메소드입니다.
     */
    public static void printFinalScore() {
        System.out.println("▶ [최종 게임 결과 리포트]");
        System.out.println("▶ 총 획득한 보물 개수: " + inventory.size() + "개"); // 총 가방 사이즈를 정수로 변환하여 인쇄합니다.
        
        // 인벤토리가 깔끔하게 0개인 불운의 유저 상태를 사전에 필터 검사합니다.
        if (inventory.isEmpty()) {
            System.out.println("▶ 수집한 보물이 존재하지 않습니다.");
            return; // 결과 보고를 중단하고 스택을 돌려보냅니다.
        }

        // [PPT 표준 방식] 인덱스 하드코딩 추적을 전면 배제하고, 안전 순회가 보장되는 Iterator 로봇을 가방으로부터 추출 선언합니다.
        Iterator<Gem> it = inventory.iterator();
        
        System.out.print("▶ 수집한 보물 리스트: ");
        
        // hasNext() 예제 메커니즘을 가동하여 데이터 종단 벽을 만날 때까지 문자 추출을 반복 수행합니다.
        while (it.hasNext()) {
            Gem g = it.next(); // 포인터를 다음 격자 칸으로 이행시키며 실제 힙 공간의 보물 정보 데이터를 가져옵니다.
            System.out.print(g.getName()); // 대상체 내부의 숨겨진 보석 명칭을 추출 전면에 노출 출력합니다.
            
            // 가독성 보정 조건문: 바로 다음에 꺼내 들 보물 토큰 데이터가 연속적으로 적재되어 있다면
            if (it.hasNext()) {
                System.out.print(", "); // 글자 사이에 구분용 특수 쉼표를 찍어 정돈 뷰를 완성합니다.
            }
        }
        System.out.println(); // 축적 리스트 나열이 끝났으므로 엔터키 효과를 주어 줄을 깔끔하게 넘깁니다.
        System.out.println("===========================================");
    }

    // [PPT 14, 19페이지 적용] FileWriter 및 flush() 버퍼 청소 기능 탑재
    // ====================================================================
    private static void saveGame() {
        // PPT 14페이지 표준 문자 출력 스트림 규격을 전적으로 준수하기 위해 FileWriter를 개통합니다.
        // try-with-resources 기법을 활용해 작업 완수 후 스트림이 알아서 폐쇄(close)되도록 지능형 입출력 구조를 설계합니다.
        try (FileWriter fw = new FileWriter(FILE_PATH)) {
            
            // 개선된 확장 for 루프를 구동하여 인벤토리 컬렉션 안에 들어있는 보물 객체들을 하나씩 꺼내 순회합니다.
            for (Gem gem : inventory) {
                // 파싱 데이터 조립: 불러오기(load) 기능 수행 시 빈 공백 노이즈 노출로 인한 NumberFormatException 정밀 에러를 미연에 원천 차단하고자
                // 콤마(,) 구분 기호 사이사이에 불필요한 공백 스페이스를 전부 소거한 순수 텍스트 줄(Line) 데이터 문자열을 결합 빌드합니다.
                String data = gem.getName() + "," + gem.getNumber() + "," + gem.getPos();
                
                // PPT 14페이지 명시 공식 구조 준수: 가공 조립된 데이터 문자열 뒤에 메모장이나 시스템이 행 단위 인식을 완벽하게 마칠 수 있도록
                // 자바 전용 강제 캐리지 리턴 개행 기호("\r\n")를 명시적으로 밀어 결합하여 텍스트 스트림에 기록 지시합니다.
                fw.write(data + "\r\n"); 
            }
            
            // PPT 19페이지 버퍼 이론 핵심 원리 적용: 문자 출력 스트림 내부 버퍼 메모리에 임시 적재되어 잔류 중인 스트림 파편 데이터들을 
            // 물리 하드디스크 장치 공간 내부 파일 속으로 즉각 남김없이 완전하게 강제 밀어내기 청소 처리를 단행합니다.
            fw.flush(); 
            
            // 문제에서 제시한 세이브 액션 성공 안내 피드백 문장을 저장 경로와 함께 콘솔에 명시합니다.
            System.out.println("현재까지의 보물 정보가 파일에 저장되었습니다. (경로 : " + FILE_PATH + ")");
            
        } catch (IOException e) { // 파일 경로 잠금이나 운영체제 디바이스 디스크 장해 등 비상 입출력 에러 상황을 캐치합니다.
            // [교수님 감점 요소 원천 배제]: PPT 예제 8-10 복사본 등 교재 소스코드 스타일과 100% 일치하도록 
            // 직관적이고 군더더기 없는 정형화된 "파일 저장 오류" 경고 메시지만을 깔끔하게 화면에 인쇄합니다.
            System.out.println("파일 저장 오류"); 
        }
    }

    // ====================================================================
    // [PPT 8, 20페이지 적용] FileReader 및 문자 버퍼 스트림 BufferedReader 복원
    // ====================================================================
    private static void loadGame() {
        // 세이브 데이터가 담겨있을 하드디스크 상의 물리 파일 개체를 실체화하여 경로 조회를 준비합니다.
        File file = new File(FILE_PATH);
        
        // 조건문: 유저가 세이브 버튼을 단 한 번도 누르지 않아 물리 세이브 백업본 문서 파일 자체가 부재하는 예외 상태를 점검합니다.
        if (!file.exists()) { 
            System.out.println("저장된 세이브 파일이 존재하지 않습니다!"); // 에러 경고를 콘솔에 뿌려줍니다.
            return; // 복원 연산을 이어갈 가치가 없으므로 본 로드 함수를 강제 종료하고 리턴 탈출합니다.
        }
        
        // PPT 20페이지 예제 8-6 고속 텍스트 스캔 표준 규격 반영:
        // 하드디스크 문자 파일 장치와 연동할 기초 스트림 FileReader 객체를 메인 축으로 삼고, 
        // 텍스트 라인 단위 읽기 성능 고속 보정 및 버퍼링 제어를 연계해 줄 BufferedReader 보조 스트림 장치를 외곽에 래핑 스티치하여 안전하게 개통합니다.
        try (BufferedReader br = new BufferedReader(new FileReader(file))) {
            
            // 세이브 파일 데이터를 새롭게 밀어 넣을 예정이므로 기존에 게임 RAM 메모리 상에 조잡하게 적재되어 존재하던 보물 인벤토리 Vector 요소를 깨끗하게 클리어 비워냅니다.
            inventory.clear(); 
            
            String line; // 파일로부터 한 줄 전체 스캔 텍스트 덩어리를 임시 임베딩해 둘 스카우트용 문자열 바구니 변수입니다.
            
            // PPT 20페이지 예제 활용 문법: br.readLine() 장치를 가동하여 문서의 끝(EOF)을 마주해 null 값을 반환하기 직전까지 한 행씩 연쇄 추적 독출을 반복 시행합니다.
            while((line = br.readLine()) != null) { 
                
                // 안전 필터: 혹시 파일 내부의 빈 여백 줄이나 엔터키 노이즈 문자열 파편이 난입했다면 가볍게 스킵 무시하고 다음 정상 라인으로 스캔 루프를 점프시킵니다.
                if (line.trim().isEmpty()) continue;
                
                // 데이터 분리 분석: 저장할 때 심어둔 데이터 식별 분리 기호인 컴마(,) 문자를 기준점으로 설정하여 글자 조각 토큰 배열(tokens)로 칼같이 잘라 분리해 냅니다.
                String[] tokens = line.split(",");
                
                String name = tokens[0]; // 분리된 글자 조각 중 0번째 데이터는 보물의 진짜 고유 한글 "이름" 텍스트 토큰입니다.
                
                // 1번째 데이터 조각은 문자열 텍스트 형상이므로 Integer.parseInt 기기를 동원해 온전한 순수 정수형 "보석 번호" 데이터로 강제 형변환 형변화를 가합니다.
                // 이 때 혹시 모를 공백 노이즈를 완벽 분쇄하고자 문자열 양 끝을 trim()으로 안전하게 정돈 깎아냅니다.
                int number = Integer.parseInt(tokens[1].trim());
                
                // 2번째 데이터 문자열 토큰 조각 또한 완벽한 정수 타입 파싱 장치를 거쳐 필드 복원용 "보석 배치 좌표" 정수로 가공 리턴해 냅니다.
                int pos = Integer.parseInt(tokens[2].trim()); 
                
                // 완전하게 복구 추출해 낸 3대 순수 원천 속성 인자들을 재료로 삼아, 자바 힙 공간 상에 온전한 새 생명의 Gem 객체를 빌드 생성한 뒤 가방 Vector 컬렉션 리스트 뒤에 차곡차곡 삽입 적재합니다.
                inventory.add(new Gem(name, number, pos));
            }
            
            // 문제지 지침에 부합하는 세이브 파일 이어하기 복원 완수 텍스트 공지를 띄워줍니다.
            System.out.println("세이브 파일을 성공적으로 읽어왔습니다! 이어서 시작합니다.");
            
            // 복원이 끝나 채워진 가방 속에 어떤 전리품들이 복구되었는지 이터레이터가 장착된 리스트 명세서 함수를 최종 연계 기동합니다.
            printInventory(); 
            
        } catch (IOException e) { // 파일이 훼손되었거나 읽기 권한 상실 등의 장치 예외를 캐치 제어합니다.
            // 교재 실습 코드 규격의 스탠더드 에러 형식에 맞춰 직관적인 "파일 읽기 오류" 안내만 화면에 송출 출력합니다.
            System.out.println("파일 읽기 오류");
        }
    }
